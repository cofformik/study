#include<iostream>
#include <vector>
using namespace std;
typedef long long LL;

const int maxn=1000010;
vector<int> ans(maxn),temp(maxn);
int n;
LL merge_sort(int l,int r){
    if(l>=r) return 0;
    LL mid=l+(r-l)/2;
    LL num=merge_sort(l,mid)+merge_sort(mid+1,r);
    int i=l,j=mid+1;
    int k=0;
    while(i<=mid&&j<=r){
        if(ans[i]<=ans[j]) temp[k++]=ans[i++];
        else{
            
            temp[k++]=ans[j++];
            num+=mid-i+1;
        }
    }
    while(i<=mid) temp[k++]=ans[i++];
    while(j<=r) temp[k++]=ans[j++];
    for(i=l,j=0;i<=r;++i,++j) ans[i]=temp[j];
    return num;
}
int main(){
    scanf("%d", &n);
    for(int i=0;i<n;++i) scanf("%d", &ans[i]);
    printf("%lld",merge_sort(0,n-1));
    return 0;
}
