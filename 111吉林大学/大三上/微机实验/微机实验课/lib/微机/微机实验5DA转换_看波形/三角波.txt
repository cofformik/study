code 	segment
		assume  cs:code
 
start:  
		mov al,00h
		;产生三角波
		mov cx, 05h     	;产生三角波的周期数为5次
aa0:
aa1:
		mov dx, 0600h
		out dx, al
		call delay      
		cmp al,0ffh
		je aa2           
		inc al            	;将al从00h步加0ffh
		jmp aa1
aa2:
		mov dx, 0600h
		out dx, al
		call delay       
		cmp al, 00h
		je aa3
		dec al            	;将al从0ffh步减至00h
		jmp aa2
aa3:
		loop aa0
 
    
delay:                
		push cx
		mov cx, 01ffh
d1: 
		loop d1
		pop cx
		ret
 
code 	ends
		end start
