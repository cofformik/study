#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
void quick_sort(vector<int>& res, int l, int r) {
    if (l >= r) return;
    int i = l - 1; int j = r + 1;
    int x = res[l + (r - l) / 2];
    while (i < j) {
        do i++; while (res[i] < x);
        do j--; while (res[j] > x);
        if (i < j) swap(res[i], res[j]);
    }
    quick_sort(res, l, j);
    quick_sort(res, j + 1, r);
}
int main() {
    int n;
    scanf("%d", &n);
    vector<int> res(n);
    for (int i = 0; i < n; ++i)
    {
        scanf("%d", &res[i]);
    }
    quick_sort(res, 0, n - 1);
    for (int k : res) printf("%d ", k);
    return 0;
}
