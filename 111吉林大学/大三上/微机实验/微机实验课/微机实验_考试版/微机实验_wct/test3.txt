	MOV DX,0642H
	MOV AL, 0FH
	OUT DX, AL

	CALL DELAY
	CALL DELAY
	MOV SIGN,00H
	IRET

DELAY PROC NEAR
	MOV CX,0FFFFH
	LOOP $
	RET
